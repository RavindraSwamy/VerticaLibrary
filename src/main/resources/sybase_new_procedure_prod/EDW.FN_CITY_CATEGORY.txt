create function EDW.FN_CITY_CATEGORY( in @CITY varchar(200) ) 
returns varchar(50)
begin
  declare @CAT varchar(50);
  set @CAT
     = case when @CITY like '%BANGALORE%'
    or @CITY like '%BENGALURU%'
    or @CITY like '%MUMBAI%'
    or @CITY like '%BOMBAY%'
    or @CITY like '%DADAR%'
    or @CITY like '%THANE%'
    or @CITY like '%NERUL%'
    or @CITY like '%MATUNGA%'
    or @CITY like '%MAHIM%'
    or @CITY like '%HYDERABAD%'
    or @CITY like '%CHENNAI%'
    or @CITY like '%MADRAS%'
    or @CITY like '%VASHI%'
    or @CITY like '%NOIDA%'
    or @CITY like '%PUNE%'
    or @CITY like '%SECUNDERABAD%'
    or @CITY like '%GURGAON%'
    or @CITY like '%KOLKATA%'
    or @CITY like '%DELHI%'
    or @CITY like '%BEGUMPET%'
    or @CITY like '%AHMEDABAD%'
    or @CITY like '%CALCUTTA%'
    or @CITY like '%NAVI MUMBAI%'
    or @CITY like '%NEW BOMBAY%'
    or @CITY like '%KOCHI%'
    or @CITY like '%COCHIN%'
    or @CITY like '%JAIPUR%'
    or @CITY like '%MOHALI%'
    or @CITY like '%CHANDIGARH%' then 'CATEGORY-I'
    when @CITY like '%BHOPAL%'
    or @CITY like '%DURGAPUR%'
    or @CITY like '%GWALIOR%'
    or @CITY like '%MYSORE%'
    or @CITY like '%MANGALORE%'
    or @CITY like '%BARELI%'
    or @CITY like '%GHAZIABAD%'
    or @CITY like '%JALGAON%'
    or @CITY like '%JAMSHEDPUR%'
    or @CITY like '%VADODARA%'
    or @CITY like '%AJMER%'
    or @CITY like '%BIKANER%'
    or @CITY like '%KOLHAPUR%'
    or @CITY like '%KANPUR%'
    or @CITY like '%AMRAVATI%'
    or @CITY like '%AMRITSAR%'
    or @CITY like '%NASHIK%'
    or @CITY like '%SALEM%'
    or @CITY like '%VIJAYAWADA%'
    or @CITY like '%KOZHIKODE%'
    or @CITY like '%GUWAHATI%'
    or @CITY like '%CHANDIGARH%'
    or @CITY like '%SILIGURI%'
    or @CITY like '%BELGAUM%'
    or @CITY like '%NAGPUR%'
    or @CITY like '%AURANGABAD%'
    or @CITY like '%SURAT%'
    or @CITY like '%JAMMU%'
    or @CITY like '%PATNA%'
    or @CITY like '%GORAKHPUR%'
    or @CITY like '%AKOLA%'
    or @CITY like '%MALEGAON%'
    or @CITY like '%KOLLAM%'
    or @CITY like '%FARIDABAD%'
    or @CITY like '%MEERUT%'
    or @CITY like '%AGRA%'
    or @CITY like '%LUDHIANA%'
    or @CITY like '%BAREILLY%'
    or @CITY like '%VARANASI%'
    or @CITY like '%TRIVANDRUM%'
    or @CITY like '%MADURAI%'
    or @CITY like '%COIMBATORE%'
    or @CITY like '%WARANGAL%'
    or @CITY like '%SOLAPUR%'
    or @CITY like '%TIRUNELVELI%'
    or @CITY like '%RANCHI%'
    or @CITY like '%NELLORE%'
    or @CITY like '%UJJAIN%'
    or @CITY like '%UDAIPUR%'
    or @CITY like '%ALLAHABAD%'
    or @CITY like '%BHAVNAGAR%'
    or @CITY like '%GULBARGA%'
    or @CITY like '%GAYA%'
    or @CITY like '%JABALPUR%'
    or @CITY like '%JALANDHAR%'
    or @CITY like '%BHUBANESHWAR%'
    or @CITY like '%GUNTUR%'
    or @CITY like '%JODHPUR%'
    or @CITY like '%INDORE%'
    or @CITY like '%MORADABAD%'
    or @CITY like '%KOTA%'
    or @CITY like '%ALIGARH%'
    or @CITY like '%BHUBANESWAR%'
    or @CITY like '%JHANSI%'
    or @CITY like '%DEHRADUN%'
    or @CITY like '%CUTTACK%'
    or @CITY like '%LUCKNOW%'
    or @CITY like '%JAMNAGAR%'
    or @CITY like '%RAJKOT%'
    or @CITY like '%ASANSOL%'
    or @CITY like '%NASIK%'
    or @CITY like '%SRINAGAR%'
    or @CITY like '%VISHAKHAPATNAM%'
    or @CITY like '%VISAKHAPATNAM%'
    or @CITY like '%VISHAKAPATNAM%'
    or @CITY like '%VIZAG%' then
      'CATEGORY-II'
    end;
  return(@CAT)
end
